1 /** @class */
  2 BetaJS.Scopes = {
  3 	
  4 	/** Takes a string and returns the global object associated with it by name.
  5      * 
  6      * @param s string (example: "BetaJS")
  7      * @param base a global namespace base (optional, will autodetect the right one if not provided)
  8      * @return global object (example: BetaJS object)
  9      */
 10 	base: function (s, base, initialize) {
 11 		if (!BetaJS.Types.is_string(s))
 12 			return s;
 13 		if (base)
 14 			return base[s];
 15 		try {
 16 			if (window && window[s])
 17 				return window[s];
 18 		} catch (e) {}
 19 		try {
 20 			if (global && global[s])
 21 				return global[s];
 22 		} catch (e) {}
 23 		try {
 24 			if (module && module.exports)
 25 				return module.exports;
 26 		} catch (e) {}
 27 		if (!initialize)
 28 		    return null;
 29         try {
 30             if (window) {
 31                 window[s] = {};
 32                 return window[s];
 33             }
 34         } catch (e) {}
 35         try {
 36             if (global) {
 37                 global[s] = {};
 38                 return global[s];
 39             }
 40         } catch (e) {}
 41         try {
 42             if (module && module.exports) {
 43                 module.exports = {};
 44                 return module.exports;
 45             }
 46         } catch (e) {}
 47         return null;
 48 	},
 49 	
 50 	/** Takes an object address string and returns the object associated with it.
 51      * 
 52      * @param s string (example: "BetaJS.Strings")
 53      * @param base a global namespace base (optional, will autodetect the right one if not provided)
 54      * @return global object (example BetaJS.Strings object)
 55      */
 56 	resolve: function (s, base) {
 57 		if (!BetaJS.Types.is_string(s))
 58 			return s;
 59 		var a = s.split(".");			
 60 		var object = this.base(a[0], base);
 61 		for (var i = 1; i < a.length; ++i)
 62 			object = object[a[i]];
 63 		return object;
 64 	},
 65 	
 66 	/** Takes an object address string and returns the object associated with it.
 67 	 *  If the address does not exist, it will be created.
 68      * 
 69      * @param s string (example: "BetaJS.Strings.FooBar")
 70      * @param base a global namespace base (optional, will autodetect the right one if not provided)
 71      * @return global object (example BetaJS.Strings.FooBar object)
 72      */
 73 	touch: function (s, base) {
 74 		if (!BetaJS.Types.is_string(s))
 75 			return s;
 76 		var a = s.split(".");		
 77 		var object = this.base(a[0], base, true);
 78 		for (var i = 1; i < a.length; ++i) {
 79 			if (!(a[i] in object))
 80 				object[a[i]] = {};
 81 			object = object[a[i]];
 82 		}
 83 		return object;
 84 	},
 85 	
 86 	/** Takes an object address string and overwrites it by a given object.
 87 	 *  If the address does not exist, it will be created.
 88      * 
 89      * @param obj object (example: {test: "foobar"})
 90      * @param s string (example: "BetaJS.Strings.FooBar")
 91      * @param base a global namespace base (optional, will autodetect the right one if not provided)
 92      * @return global object (example BetaJS.Strings.FooBar object)
 93      */
 94 	set: function (obj, s, base) {
 95 		if (!BetaJS.Types.is_string(s))
 96 			return s;
 97 		var a = s.split(".");			
 98 		var object = this.base(a[0], base, true);
 99 		for (var i = 1; i < a.length - 1; ++i) {
100 			if (!(a[i] in object))
101 				object[a[i]] = {};
102 			object = object[a[i]];
103 		}
104 		if (a.length > 1)
105 			object[a[a.length - 1]] = obj;
106 		return obj;
107 	}
108 	
109 };
